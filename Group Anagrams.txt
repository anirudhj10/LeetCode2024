Group Anagrams


class Solution(object):
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        dict1 = {}


        for str in strs:
            sorted_str = ''.join(sorted(str))
            if sorted_str not in dict1:
                dict1[sorted_str] = []
            dict1[sorted_str].append(str)
        
        return list(dict1.values())
