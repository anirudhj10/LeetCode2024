Valid Anagrams


class Solution(object):
    def isAnagram(self, s, t):
        """
        :type s: str
        :type t: str
        :rtype: bool
        """
        if(len(s) != len(t)):
            return False


        ch_map = [0] * 26


        for i in range(len(s)):
            ch_map[ord(s[i])- ord('a')] += 1
            ch_map[ord(t[i])- ord('a')] -= 1


        for count in ch_map:
            if(count != 0):
                return False


        return True
